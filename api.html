Python (requests)
import requests

data = requests.get("https://golf.propsbot.ai/golf-data.json", timeout=30).json()
print("generated:", data["generatedAt"])
print("players:  ", len(data.get("players", [])))
for p in data.get("players", [])[:5]:
    print(f"  {p.get('rank','?'):>3}  {p.get('name')}  odds={p.get('odds')}")

}

CSV Export

We only serve JSON. To get a spreadsheet-friendly player table, run this locally: 

import csv, requests

data = requests.get("https://golf.propsbot.ai/golf-data.json", timeout=30).json()
players = data.get("players", [])

# Flatten one level of nested dicts (odds, props summaries, etc.) into columns.
rows = []
for p in players:
    row = {}
    for k, v in p.items():
        if isinstance(v, (dict, list)):
            row[k] = "" if not v else str(v)  # keep simple; customize as needed
        else:
            row[k] = v
    rows.append(row)

if rows:
    cols = sorted({k for r in rows for k in r.keys()})
    with open("players.csv", "w", newline="", encoding="utf-8") as f:
        w = csv.DictWriter(f, fieldnames=cols)
        w.writeheader()
        w.writerows(rows)
    print(f"Wrote players.csv ({len(rows)} rows, {len(cols)} cols)")

Rate Limits

Served via GitHub Pages edge cache. Please cache client-side and call once per refresh interval (roughly hourly during tournaments, less often mid-week). No hard rate limit, but sustained heavy abuse may be blocked upstream by Cloudflare or GitHub.
